Quick Example

Server
let storage = SledStorageEngine::new("./data")?;
let node = Node::new_durable(Arc::new(storage), PermissiveAgent::new());

let ctx = node.context(ankurah::policy::DEFAULT_CONTEXT)?;
let trx = ctx.begin();
trx.create(&Album { 
    name: "Parade".into(),
    year: 1986 
}).await?;
trx.commit().await?;
Client
let storage = IndexedDBStorageEngine::open("myapp").await?;
let node = Node::new(Arc::new(storage), PermissiveAgent::new());
let client = WebsocketClient::new(node.clone(), "ws://localhost:9797")?;

let ctx = node.context(ankurah::policy::DEFAULT_CONTEXT)?;
let albums = ctx.query::<AlbumView>("year > 1985").await?;
React Component
export const AlbumList = signalObserver(() => {
  const albums = useMemo(() => Album.query(ctx(), "year > 1985"), []);
  
  return (
    <ul>
      {albums.items.map(album => (
        <li key={album.id}>{album.name} ({album.year})</li>
      ))}
    </ul>
  ); // albums.items is tracked by signalObserver
});
